# For AboutDialog.h
# Add a target that will provide the git revision info
# whenever it changes.  (Must be done at build time, not
# configuration time.)
def_vars()
if( GIT_FOUND )
   add_custom_target(
      version
      COMMAND
         ${CMAKE_COMMAND} -D GIT=${GIT_EXECUTABLE}
                          -D _PRVDIR=${_PRVDIR}
                          -P ${CMAKE_MODULE_PATH}/Version.cmake
      WORKING_DIRECTORY
         ${topdir}
   )
   add_dependencies( etc version )
endif()
